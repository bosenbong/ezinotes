// 🎮 FEVER 3D - Build it Yourself! // Step 1: 3D World with WASD let gameStarted = false; // Camera let camX = 0, camY = 50, camZ = 0; let yaw = 0, pitch = 0; // Player physics let posX = 0, posY = 0, posZ = 0; let velY = 0; let gravity = 0.5; let jumpPower = 12; let moveSpeed = 8; let grounded = false; // Game state let level = 1; let deaths = 0; let fever = 0; let message = ""; let messageTimer = 0; // World let platforms = []; let holes = []; let traps = []; let goal = null; const keys = {}; function setup() { createCanvas(800, 600, WEBGL); // Controls document.addEventListener('keydown', e => keys[e.key.toLowerCase()] = true); document.addEventListener('keyup', e => keys[e.key.toLowerCase()] = false); generateLevel(1); } function draw() { background(fever > 30 ? 30 : 20, 20, 30); if (!gameStarted) { drawStartScreen(); return; } // Lighting ambientLight(80); directionalLight(255, 200, 200, 0.5, -1, 0.5); // Camera camera(0, 0, 0, 0, 0, -1, 0, 1, 0); // Apply camera rotation rotateY(yaw); rotateX(pitch); // Move player updatePlayer(); // Draw world drawWorld(); // UI (reset transform) resetMatrix(); drawUI(); // Fever effects if (fever > 40 && random() < 0.1) { background(255, 0, 0, 30); } } function drawStartScreen() { textAlign(CENTER); textSize(60); fill(255, 50, 50); text("FEVER 3D", 0, -50); textSize(20); fill(200); text("A game that hates you", 0, 0); textSize(16); fill(150); text("WASD to move | Mouse to look | Space to jump", 0, 50); text("Click to start", 0, 100); } function updatePlayer() { // WASD movement (relative to camera direction) let forward = 0; let right = 0; if (keys['w']) forward = moveSpeed; if (keys['s']) forward = -moveSpeed; if (keys['a']) right = -moveSpeed; if (keys['d']) right = moveSpeed; // Move in facing direction posX += Math.sin(yaw) * forward + Math.cos(yaw) * right; posZ += Math.cos(yaw) * forward - Math.sin(yaw) * right; // Jump if (keys[' '] && grounded) { velY = -jumpPower; grounded = false; if (random() < 0.2) showMessage("NOPE LMAO"); } // Gravity velY += gravity; posY += velY; // Collision with platforms grounded = false; for (let p of platforms) { if (posX > p.x - p.w/2 && posX < p.x + p.w/2 && posZ > p.z - p.d/2 && posZ < p.z + p.d/2 && posY < p.y && posY > p.y - 30 && velY >= 0) { posY = p.y; velY = 0; grounded = true; } } // Check holes for (let h of holes) { if (Math.abs(posX - h.x) < 30 && Math.abs(posZ - h.z) < 30 && posY < 10) { die("FELL IN HOLE!"); } } // Check traps for (let t of traps) { if (Math.abs(posX - t.x) < 30 && Math.abs(posZ - t.z) < 30 && posY < 10) { die("TRAP!"); } } // Fall death if (posY > 200) { die("FELL OFF!"); } // Goal if (goal && Math.abs(posX - goal.x) < 40 && Math.abs(posZ - goal.z) < 40 && posY < goal.y + 20) { winLevel(); } } function drawWorld() { // Draw platforms for (let p of platforms) { push(); translate(p.x, p.y, p.z); if (p.type === "normal") fill(100, 150, 255); else if (p.type === "fake") fill(100, 200, 100); // Looks safe! else if (p.type === "moving") fill(150, 100, 255); else if (p.type === "goal") fill(255, 220, 0); box(p.w, 15, p.d); pop(); // Moving platforms if (p.type === "moving") { p.x += Math.sin(frameCount * 0.03) * 2; } } // Draw holes (as dark spots) noStroke(); fill(0); for (let h of holes) { push(); translate(h.x, 5, h.z); ellipse(0, 0, 60, 60); pop(); } // Draw traps fill(255, 50, 100); for (let t of traps) { push(); translate(t.x, 5, t.z); ellipse(0, 0, 50, 50); pop(); } // Draw goal if (goal) { push(); translate(goal.x, goal.y, goal.z); fill(255, 220, 0); box(60, 10, 60); pop(); } } function drawUI() { // Fever bar fill(50); rect(-380, -280, 200, 20); fill(255, 50, 50); rect(-380, -280, fever * 4, 20); fill(255); textSize(14); text("FEVER: " + Math.floor(fever * 2) + "%", -375, -265); // Level & deaths textSize(16); textAlign(RIGHT); text("Level " + level, 380, -270); text("Deaths: " + deaths, 380, -250); // Message if (messageTimer > 0) { messageTimer--; textAlign(CENTER); textSize(28); fill(255, 255, 0); text(message, 0, -100); } } function generateLevel(lvl) { platforms = []; holes = []; traps = []; // Starting platform platforms.push({x: 0, y: 0, z: 0, w: 150, d: 100, type: "normal"}); // Generate path let lastX = 0; let lastZ = -80; let lastY = 0; for (let i = 0; i < 10 + lvl * 2; i++) { lastX += random(-100, 100); lastZ -= random(80, 150); lastY -= random(20, 60); let type = "normal"; if (random() < 0.15 + lvl * 0.03) type = "fake"; if (random() < 0.1 + lvl * 0.05) type = "moving"; platforms.push({ x: lastX, y: lastY, z: lastZ, w: random(60, 100), d: random(60, 100), type: type }); // Add holes around if (random() < 0.3) { holes.push({x: lastX + random(-50, 50), z: lastZ + random(-50, 50)}); } // Add traps if (random() < 0.1 + lvl * 0.05) { traps.push({x: lastX + random(-80, 80), z: lastZ + random(-50, 50)}); } } // Goal goal = {x: lastX, y: lastY + 30, z: lastZ - 50}; } function die(reason) { deaths++; fever = min(fever + 5, 50); showMessage(random([ "L + RATIO", "SKILL ISSUE", "GIT GUD", "CRINGE", "NOT LIKE THIS", "💀", "EZ", "LOOOL" ])); // Reset position posX = 0; posY = 0; posZ = 0; velY = 0; } function winLevel() { level++; showMessage("LUCKY!"); generateLevel(level); posX = 0; posY = 0; posZ = 0; } function showMessage(msg) { message = msg; messageTimer = 120; } function mousePressed() { if (!gameStarted) { gameStarted = true; generateLevel(1); } } function mouseMoved() { if (gameStarted) { yaw -= movedX * 0.005; pitch -= movedY * 0.005; pitch = constrain(pitch, -PI/3, PI/3); } }
